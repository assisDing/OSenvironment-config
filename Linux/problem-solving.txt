              #!/bin/bash
              
              # i3lock blurred screen inspired by /u/patopop007 and the blog post
              # http://plankenau.com/blog/post-10/gaussianlock
              
              # Timings are on an Intel i7-2630QM @ 2.00GHz
              
              # Dependencies:
              # imagemagick
              # i3lock
              # scrot (optional but default)
              
              IMAGE=/tmp/i3lock.png
              SCREENSHOT="scrot $IMAGE" # 0.46s
              
              # Alternate screenshot method with imagemagick. NOTE: it is much slower
              # SCREENSHOT="import -window root $IMAGE" # 1.35s
              
              # Here are some imagemagick blur types
              # Uncomment one to use, if you have multiple, the last one will be used
              
              # All options are here: http://www.imagemagick.org/Usage/blur/#blur_args
              #BLURTYPE="0x5" # 7.52s
              #BLURTYPE="0x2" # 4.39s
              #BLURTYPE="5x2" # 3.80s
              BLURTYPE="2x8" # 2.90s
              #BLURTYPE="2x3" # 2.92s
              
              # Get the screenshot, add the blur and lock the screen with it
              $SCREENSHOT
              convert $IMAGE -blur $BLURTYPE $IMAGE
              i3lock -i $IMAGE
              rm $IMAGE
              nao fazer nada quando fechar a tela
              sudo nano /etc/systemd/logind.conf
              
              HandleLidSwitch=ignore
              HandleLidSwitchDocked=ignore
              HandleLidSwitchExternalPower=ignore
              HandleLidSwitchDocked=ignore
              
              
    sudo systemctl restart systemd-logind             
    i3-msg reload
    i3-msg restart

	n    cria nova partição

	w    salva e sai
	
	p    mostra as partições atuais (opcional, só para conferir)
	d    apaga uma partição (repita até apagar todas)
	g    cria nova tabela de partição GPT (disco limpo)
	n    cria nova partição primária
	Enter    aceita início padrão
	Enter    usa o tamanho total do disco
	w    salva e sai

#evitar super aquecimento
	#Desativar modo turbo
		echo 1 | sudo tee /sys/devices/system/cpu/intel_pstate/no_turbo

	#Ativar modo turbo
		echo 0 | sudo tee /sys/devices/system/cpu/intel_pstate/no_turbo

	#Ative economia
		echo powersave | sudo tee /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor

	#verifique
		#cat /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor | sort -u

	#Se quiser voltar a performance maxima
		echo performance | sudo tee /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor


	#Listar esquemas de performance suportados
		cat /sys/devices/system/cpu/cpu*/cpufreq/scaling_available_governors | sort -u
